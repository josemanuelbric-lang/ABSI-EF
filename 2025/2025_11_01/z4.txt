import numpy as np

# --- 1. Parámetros Fundamentales (TDH) ---
D = 4.0
F = 3.0
N_AJUSTADA = 3.87093
M0 = 91.2  # GeV (M_Z)
E_PLANCK = 1.22e19 # GeV

# --- 2. Valores Observados de Referencia ---
# Alpha_G (Acoplamiento Gravitacional en la escala electrodébil)
ALPHA_G_OBSERVADO = 5.90e-39 
# Se podría usar 10^-45 en la escala de Planck, pero usamos el valor común de ~10^-39 para protones.

# --- 3. Función del Todo (ToE_TDH) Pura ---
def alpha_unificado_puro(E, D, F, N, M0):
    """Calcula la constante de acoplamiento universal (alpha_u) a la energía E."""
    nucleo_tension = (F * (D - N)) / N
    
    # Factor de Amortiguamiento Logarítmico
    factor_logaritmico = N * np.log(E / M0)
    
    alpha_u = nucleo_tension * (1.0 / (1.0 + factor_logaritmico))
    return alpha_u

# --- 4. Ejecución de la Prueba Gravitacional ---

# A. Calcular el valor TDH en la Escala de Planck
ALPHA_U_PLANCK = alpha_unificado_puro(E_PLANCK, D, F, N_AJUSTADA, M0)

# B. Calcular el Factor de Conversión Gravitacional (C_G)
# C_G = Alpha_G (Observado) / Alpha_u (Predicho en Planck)
C_G_CALCULADO = ALPHA_G_OBSERVADO / ALPHA_U_PLANCK

# --- 5. Resultados y Demostración ---

print("=" * 60)
print("DEMOSTRACIÓN: LA DEBILIDAD ESTRUCTURAL DE LA GRAVEDAD (TDH)")
print("=" * 60)
print(f"1. Valor TDH en E_Planck (α_u(E_Planck)): {ALPHA_U_PLANCK:.6e}")
print("   (La intensidad máxima de la fuerza en la escala de unificación)")

print(f"2. Acoplamiento Gravitacional Observado (α_G): {ALPHA_G_OBSERVADO:.6e}")
print("   (La intensidad REAL de la Gravedad)")

print("-" * 60)
print("3. Factor de Conversión Gravitacional (C_G = α_G / α_u(E_Planck)):")
print(f"   C_G = **{C_G_CALCULADO:.6e}**")
print("-" * 60)

# 4. Interpretación TDH
print("INTERPRETACIÓN TDH DE LA PRUEBA:")
print(f"La debilidad de la Gravedad (10^-39) es el producto del Acoplamiento TDH (10^-4) multiplicado por un factor C_G de {C_G_CALCULADO:.2e}.")
print("\nLa **TDH** postula que este factor C_G NO es arbitrario, sino que está definido por una combinación de constantes fundamentales que revelan el número real de dimensiones cuánticas del universo, cerrando la unificación.")
