import math

def segmentar_negativos_lista(numeros):
    lista_descenso_R = []
    lista_ascenso_L = []
    for n in numeros:
        if n % 2 == 0:
            # Regla de Descenso R (aplicada a pares)
            resultado = n // 2
            lista_descenso_R.append(resultado)
        else:
            # Regla de Ascenso L (aplicada a impares)
            resultado = n * 3 + 1
            lista_ascenso_L.append(resultado)
    return lista_descenso_R, lista_ascenso_L
limite_test=100
mi_lista_negativa = list(range(-1, -(limite_test + 1), -1))
pares_R, impares_L = segmentar_negativos_lista(mi_lista_negativa)

print(pares_R)

print(impares_L)
