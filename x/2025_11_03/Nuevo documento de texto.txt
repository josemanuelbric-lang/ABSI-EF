def es_primo(n):
    """Verifica si un n칰mero es primo."""
    if n < 2:
        return False
    for i in range(2, int(np.sqrt(n)) + 1):
        if n % i == 0:
            return False
    return True

def encontrar_pares_goldbach(numero_par):
    """Simula la demostraci칩n TDH-TN: busca el par primo sim칠trico."""
    if numero_par % 2 != 0 or numero_par <= 2:
        return None, None, None

    k = numero_par // 2
    delta = 0

    # Buscamos la '칩rbita' de primos alrededor de k
    while True:
        p1 = k + delta
        p2 = k - delta

        if p2 <= 1:
            # En la pr치ctica formal, esto nunca deber칤a pasar para pares > 4
            # Si p2 llega a 1, Goldbach no se cumple (Teor칤a fallida).
            return None, None, None

        es_p2_primo = es_primo(p2)

        if es_p1_primo and es_p2_primo:
            # 칄xito: Encontramos el par sim칠trico (La ausencia de N garantiz칩 la cobertura)
            return p1, p2, delta

        # Aumentamos la 'distancia' (delta) de la simetr칤a k
        delta += 1
